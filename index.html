function hitungTarif() { const jarak = parseFloat(document.getElementById("jarak").value); const parkir = document.getElementById("parkir").checked; const cuaca = document.querySelector('input[name="cuaca"]:checked').value; const move = document.querySelector('input[name="move"]:checked').value; const ramai = document.querySelector('input[name="ramai"]:checked').value; // ‚Üê tambahkan ini if (isNaN(jarak) || jarak < 0) { document.getElementById("hasil").textContent = "Masukkan jarak yang valid!"; return; } let total = 0; if (jarak <= 2) { total = 4000; } else if (jarak > 2 && jarak < 3) { total = 5500; } else if (jarak >= 3 && jarak <= 5.5) { let tambahan = Math.ceil((jarak - 3) / 1) * 1500; total = 5500 + tambahan; } else if (jarak > 5.5 && jarak <= 10) { let tambahan = Math.ceil((jarak - 5.5) / 0.5) * 1500; total = 5500 + (3 * 1500) + tambahan; } else { total = 15000; } if (parkir) { total += 2000; } if (cuaca === "ringan") { total += 1000; } else if (cuaca === "deras") { total += 2000; } if (move === "pindah") { total += 1000; } if (ramai === "ramai") { total += 2000; } document.getElementById("hasil").textContent = `Total Tarif: Rp${total.toLocaleString()}`; }
